int binarySearch(int[] numbers, int M) {
  Arrays.sort(numbers);

  int left = 0;
  int right = numbers.length - 1;
  int mid; 

  while (left < right) {
    mid = (left + right) / 2;

    if (numbers[mid] == M)
      return mid;
    else if (numbers[mid] < M) 
      left = mid + 1;
    else 
      right = mid - 1;
  }
}
